#include <stdio.h>
int main(void){
    int i = 1;
    while (i < 100) {
        if (i % 3 == 0 || i % 5 == 0) {
            if (i % 3 == 0) printf("Fizz");
            if (i % 5 == 0) printf("Buzz");
            puts("!");
        }
        else {
            printf("%i\n", i);
        }
        i++;
    }
    return 0;
}
